CREATE DATABASE EnglishAppV1
GO
USE EnglishAppV1
GO
-- Bảng Roles
CREATE TABLE Roles (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name NVARCHAR(10)
);

-- Bảng Users
CREATE TABLE Users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username NVARCHAR(50),
  email NVARCHAR(100),
  password NVARCHAR(200),
  avatar NVARCHAR(200),
  isUserLocked BIT
);

-- Bảng UserRole
CREATE TABLE UserRole (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  role_id INT,
  FOREIGN KEY (user_id) REFERENCES Users(id),
  FOREIGN KEY (role_id) REFERENCES Roles(id)
);

-- Bảng Vocabulary
CREATE TABLE Vocabulary (
  id INT AUTO_INCREMENT PRIMARY KEY,
  english NVARCHAR(200),
  vietnamese NVARCHAR(200),
  description NVARCHAR(500),
  user_id INT,
  FOREIGN KEY (user_id) REFERENCES Users(id)
);
